# Raid Bot Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Windows Python daemon that watches Telegram as a user account, detects new Raidar raid posts in whitelisted chats, and opens each matching X status URL once per process in a specific Chrome profile.

**Architecture:** Use a small Python package with clear boundaries: config loading, raid parsing, in-memory dedupe, Chrome launching, message processing, and a Telethon listener. Keep the business logic synchronous and deterministic where possible, then wrap it in a thin async Telegram adapter so most behavior can be covered with fast unit tests.

**Tech Stack:** Python 3.12, Telethon, python-dotenv, pytest, pytest-asyncio, standard-library `logging`, `subprocess`, `asyncio`, `pathlib`, and `re`

## Implementation Notes

- Treat a URL as "opened" only after `subprocess.Popen(...)` succeeds without raising. This allows later duplicate Raidar updates to retry if Chrome launch failed the first time.
- Match the sender by Telegram numeric ID, not by display name.
- Normalize `twitter.com` and `x.com` URLs into a single canonical `https://x.com/<user>/status/<id>` form before dedupe.
- Reject queue posts by checking for `Next up...` before accepting raid markers.
- Process only new incoming messages. Do not backfill missed history on startup.

```python
from dataclasses import dataclass
from pathlib import Path
import os


@dataclass(frozen=True)
class Settings:
    telegram_api_id: int
    telegram_api_hash: str
    telegram_session_path: Path
    telegram_chat_whitelist: set[int]
    raidar_sender_id: int
    chrome_path: Path
    chrome_user_data_dir: Path
    chrome_profile_directory: str
    open_cooldown_seconds: float = 0.0
    log_level: str = "INFO"

    @classmethod
    def from_env(cls) -> "Settings":
        return cls(
            telegram_api_id=int(_require("TELEGRAM_API_ID")),
            telegram_api_hash=_require("TELEGRAM_API_HASH"),
            telegram_session_path=Path(_require("TELEGRAM_SESSION_PATH")),
            telegram_chat_whitelist={int(value) for value in _require("TELEGRAM_CHAT_WHITELIST").split(",")},
            raidar_sender_id=int(_require("RAIDAR_SENDER_ID")),
            chrome_path=Path(_require("CHROME_PATH")),
            chrome_user_data_dir=Path(_require("CHROME_USER_DATA_DIR")),
            chrome_profile_directory=_require("CHROME_PROFILE_DIRECTORY"),
            open_cooldown_seconds=float(os.getenv("OPEN_COOLDOWN_SECONDS", "0")),
            log_level=os.getenv("LOG_LEVEL", "INFO").upper(),
        )


def _require(name: str) -> str:
    value = os.getenv(name)
    if value is None or not value.strip():
        raise ValueError(f"Missing required setting: {name}")
    return value.strip()
```

```python
from dataclasses import dataclass
import re


STATUS_URL_RE = re.compile(r"https?://(?:www\.)?(?:x|twitter)\.com/([^/\s]+)/status/(\d+)", re.IGNORECASE)
RAID_MARKERS = ("likes", "retweets", "replies")


@dataclass(frozen=True)
class RaidMatch:
    raw_url: str
    normalized_url: str


def parse_raid_message(text: str) -> RaidMatch | None:
    lowered = text.lower()
    if "next up..." in lowered:
        return None
    if not any(marker in lowered for marker in RAID_MARKERS):
        return None

    url_match = STATUS_URL_RE.search(text)
    if not url_match:
        return None

    user, status_id = url_match.groups()
    return RaidMatch(
        raw_url=url_match.group(0),
        normalized_url=f"https://x.com/{user}/status/{status_id}",
    )
```

```python
class InMemoryOpenedUrlStore:
    def __init__(self) -> None:
        self._opened_urls: set[str] = set()

    def contains(self, url: str) -> bool:
        return url in self._opened_urls

    def mark_if_new(self, url: str) -> bool:
        if url in self._opened_urls:
            return False
        self._opened_urls.add(url)
        return True
```

```python
from pathlib import Path
import subprocess


class ChromeOpener:
    def __init__(self, chrome_path: Path, user_data_dir: Path, profile_directory: str, launcher=None) -> None:
        self.chrome_path = chrome_path
        self.user_data_dir = user_data_dir
        self.profile_directory = profile_directory
        self.launcher = launcher or subprocess.Popen

    def open(self, url: str) -> None:
        command = [
            str(self.chrome_path),
            "--new-tab",
            f"--user-data-dir={self.user_data_dir}",
            f"--profile-directory={self.profile_directory}",
            url,
        ]
        self.launcher(command)
```

```python
from raidbot.models import MessageOutcome
from raidbot.parser import parse_raid_message


class RaidService:
    def __init__(self, allowed_chat_ids: set[int], allowed_sender_id: int, opener, dedupe_store) -> None:
        self.allowed_chat_ids = allowed_chat_ids
        self.allowed_sender_id = allowed_sender_id
        self.opener = opener
        self.dedupe_store = dedupe_store

    def handle_message(self, message):
        if message.chat_id not in self.allowed_chat_ids:
            return MessageOutcome(action="skipped", reason="chat_not_whitelisted")
        if message.sender_id != self.allowed_sender_id:
            return MessageOutcome(action="skipped", reason="sender_not_allowed")

        match = parse_raid_message(message.text)
        if match is None:
            return MessageOutcome(action="skipped", reason="not_a_raid")
        if self.dedupe_store.contains(match.normalized_url):
            return MessageOutcome(action="skipped", reason="duplicate", normalized_url=match.normalized_url)

        self.opener.open(match.normalized_url)
        self.dedupe_store.mark_if_new(match.normalized_url)
        return MessageOutcome(action="opened", reason="raid_opened", normalized_url=match.normalized_url)
```

```python
from telethon import TelegramClient, events

from raidbot.models import IncomingMessage


def event_to_incoming_message(event) -> IncomingMessage:
    return IncomingMessage(
        chat_id=event.chat_id,
        sender_id=event.sender_id,
        text=event.raw_text or "",
    )


class TelegramRaidListener:
    def __init__(self, api_id: int, api_hash: str, session_path: str, on_message) -> None:
        self.client = TelegramClient(session_path, api_id, api_hash)
        self.on_message = on_message

    async def run_forever(self) -> None:
        @self.client.on(events.NewMessage(incoming=True))
        async def _handle(event):
            self.on_message(event_to_incoming_message(event))

        await self.client.start()
        await self.client.run_until_disconnected()
```

```python
from dataclasses import dataclass

from raidbot.chrome import ChromeOpener
from raidbot.dedupe import InMemoryOpenedUrlStore
from raidbot.service import RaidService
from raidbot.telegram_client import TelegramRaidListener


@dataclass
class Runtime:
    service: RaidService
    listener: TelegramRaidListener


def build_runtime(settings):
    opener = ChromeOpener(
        chrome_path=settings.chrome_path,
        user_data_dir=settings.chrome_user_data_dir,
        profile_directory=settings.chrome_profile_directory,
    )
    service = RaidService(
        allowed_chat_ids=settings.telegram_chat_whitelist,
        allowed_sender_id=settings.raidar_sender_id,
        opener=opener,
        dedupe_store=InMemoryOpenedUrlStore(),
    )
    listener = TelegramRaidListener(
        api_id=settings.telegram_api_id,
        api_hash=settings.telegram_api_hash,
        session_path=str(settings.telegram_session_path),
        on_message=service.handle_message,
    )
    return Runtime(service=service, listener=listener)
```
